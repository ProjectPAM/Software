305 {
306  qint64 maxSize = bytesAvailable_sys();
307  if (maxSize > 0) {
308  char * writePtr = readBuffer.reserve(size_t(maxSize));
309  qint64 bytesRead = readData_sys(writePtr, maxSize);
310  if (bytesRead < maxSize)

979 qint64 QextSerialPort::readData(char *data, qint64 maxSize)
980 {
981  Q_D(QextSerialPort);
982  QWriteLocker locker(&d->lock);
983  qint64 bytesFromBuffer = 0;
984  if (!d->readBuffer.isEmpty()) {
985  bytesFromBuffer = d->readBuffer.read(data, maxSize);
986  if (bytesFromBuffer == maxSize)
987  return bytesFromBuffer;
988  }
989  qint64 bytesFromDevice = d->readData_sys(data+bytesFromBuffer, maxSize-bytesFromBuffer);
990  if (bytesFromDevice < 0) {
991  return -1;
992  }
993  return bytesFromBuffer + bytesFromDevice;
994 }
