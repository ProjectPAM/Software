164 void CrushedBitMap::inflateSlice(QImage* pImage, int xOffset, int yOffset, bool bUseNaturalSize)
165 {
166  if(pImage == NULL) return; // No image to draw
167 
168  // Not using the slice's natural size, so pImage should have some size greater than zero!
169  if(!bUseNaturalSize && (pImage->width()<=0 || pImage->height()<=0))
170  return;
171 
172 
173  mIndex = 0; // Reset to start
174  bool bCurColorIsWhite = false;
175  unsigned uCurrentPos = 0;
176  unsigned uImageSize = 0;
177  unsigned uData = 0;
178  int iKey;
179 
180  if(!m_bIsBaseLayer){
181  // not a fake base layer, so inflate the natural width and height info and the first color (white or not?)
182  m_iWidth = popBits(16);
183  m_iHeight = popBits(16);
184  bCurColorIsWhite = (popBits(1)==1);
185  uImageSize = m_iWidth * m_iHeight;
186  }
187 
188  if(bUseNaturalSize)
189  {
190  if(m_iWidth<1 || m_iHeight<1)
191  {
192  m_iWidth = pImage->width();
193  m_iHeight = pImage->height();
194  *pImage = QImage(m_iWidth,m_iHeight,QImage::Format_ARGB32_Premultiplied);
195  pImage->fill(qRgba(0,0,0,255));
196  return;
197  }
198  // Using the slice's natural size, so we create a new QImage of this size and fill it with black pixels
199  *pImage = QImage(m_iWidth,m_iHeight,QImage::Format_ARGB32_Premultiplied);
200  pImage->fill(qRgba(0,0,0,255));
201  }
202 
203  // note, WinXXX will equal SliceXXX if natural sized
204  float WinWidth = pImage->width()/2.0;
205  float WinHeight= pImage->height()/2.0;
206  float SliceWidth = m_iWidth/2.0;
207  float SliceHeight = m_iHeight/2.0;
208  // Find centering offsets (zero if equal sizes
209  int widthOff = (int)(WinWidth - SliceWidth);
210  int heightOff = (int)(WinHeight - SliceHeight);
211  // Find adjusted offsets (from centered) based on input parameters
212  m_xOffset = widthOff + xOffset;
213  m_yOffset = heightOff + yOffset;
214 
215  if(m_bIsBaseLayer) return; // If it's a base layer, we're done (no "non-support" image to inflate)
216 
217  // inflate the first pixel run info (uData = run length)
218  iKey = popBits(5);
219  uData = popBits(iKey+1);
220 
221  while((uData > 0) && (uCurrentPos < uImageSize)) { //if uData <= 0, we're done. if uCurrentPos >= uImage Size we've set the last pixel
222  for(unsigned ui=0; ui<uData; ui++) {
223  if(bCurColorIsWhite) setWhiteImagePixel(pImage, uCurrentPos); // Set the corresponding pixel in pImage
224  uCurrentPos++; //Note we could exceed uImageSize but setWhiteImagePixel above will safely ignore this error
225  }
226  // pop length of next run of pixels
227  iKey = popBits(5);
228  uData = popBits(iKey+1);
229  // toggle current color
230  if(bCurColorIsWhite) bCurColorIsWhite = false; else bCurColorIsWhite = true;
231  }
232 
233  return;
234 }

259 bool CrushedBitMap::crushSlice(QImage* pImage)
260 {
261  if(pImage == NULL) return false;
262 
263  m_iWidth=pImage->width();
264  m_iHeight=pImage->height();
265 
266  unsigned uCurrentPos = 0;
267  unsigned uImageSize = m_iWidth * m_iHeight;
268  unsigned uData = 0;
269  int iKey;
270  uiWhitePixels = 0;
271  bool bCurColorIsWhite = pixelIsWhite(pImage, 0);
272 
273  // reset the bit array
274  if (mBitarray.size()>0) mBitarray.resize(0);
275  mIndex = 0;
276 
277  // reset extents
278  mExtents.setBottomRight(QPoint(0,0));
279  mExtents.setTopLeft(QPoint(pImage->width(),pImage->height()));
280 
281  // push the image width and height
282  pushBits(pImage->width(),16);
283  pushBits(pImage->height(),16);
284 
285  // push the first colorbit (0 or 1)
286  pushBits(bCurColorIsWhite, 1);
287 
288  // loop through all the pixels in the image
289  do {
290  uData = 0;
291  bCurColorIsWhite = pixelIsWhite(pImage, uCurrentPos); //returns true if pixel is white, updates extents
292  // count the pixels until the color changes
293  while ((uCurrentPos < uImageSize) && (bCurColorIsWhite == pixelIsWhite(pImage, uCurrentPos))) {
294  uCurrentPos++;
295  uData++;
296  if(bCurColorIsWhite)uiWhitePixels++;
297  }
298  // store the count
299  iKey = computeKeySize(uData);
300  if(iKey<0) return false;
301  pushBits(iKey, 5);
302  pushBits(uData, iKey+1);
303  } while (uCurrentPos < uImageSize);
304  return true;
305 }

307 bool CrushedBitMap::pixelIsWhite(QImage* pImage, unsigned uCurPos)
308 {
309  // define a black pixel
310  //QRgb blackPixel = QColor(0,0,0).rgb();
311  int x, y;
312  y = uCurPos / pImage->width();
313  x = uCurPos - y*pImage->width();
314  // Import "mostly black" pixels as black, else white. This helps clean up .jpg compression artifacts.
315  QColor c = pImage->pixel(x,y);
316  if(c.red()<32 && c.blue()<32 && c.green()<32) return false;
317 
318  if(x < mExtents.left() ) mExtents.setLeft(x);
319  if(x > mExtents.right() ) mExtents.setRight(x);
320  if(y > mExtents.bottom()) mExtents.setBottom(y);
321  if(y < mExtents.top() ) mExtents.setTop(y);
322 
323  return true;

335 void CrushedBitMap::pushBits(int iValue, int iBits)
336 {
337  // Pushes the iBits least significant bits of iValue into the bit array
338  // iBits should always be smaller than 33
339  if (mIndex + iBits > mBitarray.size()) mBitarray.resize(mIndex + iBits);
340  for (int i=iBits-1; i>=0; i--) {
341  mBitarray.setBit(mIndex, iValue & (int)qPow(2.0,i));
342  mIndex++;
343  }
344 }

346 int CrushedBitMap::popBits(int iBits)
347 {
348  // Pops iBits from the array (up to 32) returns -1 if iBits>32 or end of array reached
349  int iReturn = -1;
350  if (mIndex + iBits <= mBitarray.size() && iBits<33){
351  iReturn = 0;
352  for (int i=iBits-1; i>=0; i--) {
353  iReturn = (iReturn<<1) + mBitarray.testBit(mIndex);
354  mIndex++;
355  }
356  }
357  return iReturn;
358 }
