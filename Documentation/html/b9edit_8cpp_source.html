366  //do a quick search for the word "slic3r:type="contour"" and then the following "style="fill:"
367  //to figure out whether the image is inverted or not.
368  QString buff = "";

370  searchfile.open(QIODevice::ReadOnly);
371  QTextStream searchstream(&searchfile);
372 
373 
374  while(buff != "slic3r:type=\"contour\"" && !searchstream.atEnd())
375  {
376  searchstream >> buff;
377  }
378  if(!searchstream.atEnd())//we must have found it.
379  {
380  while(buff != "style=\"fill:" && !searchstream.atEnd())
381  {
382  searchstream >> buff;
383  }

385  {
386  searchstream >> buff;
387  if(buff == "white\"")
388  {
389  inverted = false;

394  }

407  QMessageBox msgBox;

411  return;

421  y = SvgRender.boundsOnElement("layer" + QString().number(layers)).bottom();
