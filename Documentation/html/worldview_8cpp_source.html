413 void WorldView::ResetCamera(bool xrayon)
414 {
415  double nearClip;
416  //perpective/ortho/xray stuff!
417  glMatrixMode(GL_PROJECTION);
418  glLoadIdentity();//restores default matrix.
419  if(perspective)
420  {
421  if(pMain->XRayEnabled() && pMain->SupportModeInst() && xrayon)
422  {
423  nearClip = camdist -
424  (pMain->SupportModeInst()->GetMaxBound() - pMain->SupportModeInst()->GetMinBound()).length()*(1-pMain->XRayPercentage())
425  +(pMain->SupportModeInst()->GetMaxBound() - pMain->SupportModeInst()->GetMinBound()).length()*0.5;
426  }
427  else
428  nearClip = 1;
429 
430  gluPerspective(30,double(width())/height(),nearClip,5500);
431  }
432  else
433  {
434  float aspRatio = float(this->width())/this->height();
435  if(pMain->XRayEnabled() && pMain->SupportModeInst() && xrayon)
436  {
437  nearClip = 200-
438  (pMain->SupportModeInst()->GetMaxBound() - pMain->SupportModeInst()->GetMinBound()).length()*(1-pMain->XRayPercentage())
439  +(pMain->SupportModeInst()->GetMaxBound() - pMain->SupportModeInst()->GetMinBound()).length()*0.5;
440  }
441  else
442  nearClip = 0.1;
443 
444  glOrtho(-(camdist/120.0)*aspRatio*102.4/2,
445  (camdist/120.0)*aspRatio*102.4/2,
446  -(camdist/120.0)*102.4/2,
447  (camdist/120.0)*102.4/2,
448  nearClip,
449  5500);
450  }
451  glMatrixMode(GL_MODELVIEW);
452  glLoadIdentity();
453 
454  if(perspective)
455  glTranslatef(0.0, 0.0, -camdist);//step back amount
456  else
457  glTranslatef(0.0, 0.0, -200.0);//step back amount
458 
459  glTranslatef(pan.x(),-pan.y(),0);
460 
461  glRotatef(xRot, 1.0, 0.0, 0.0);
