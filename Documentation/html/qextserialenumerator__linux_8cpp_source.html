70 QList<QextPortInfo> QextSerialEnumeratorPrivate::getPorts_sys()
71 {
72  QList<QextPortInfo> returnVal;
73  struct udev *ud = udev_new();
74  if (!ud) {
75  qCritical() << "Unable to enumerate ports because udev is not initialized.";
76  return returnVal;
77  }
78 
79  struct udev_enumerate *enumerate = udev_enumerate_new(ud);
80  udev_enumerate_add_match_subsystem(enumerate, "tty");
81  udev_enumerate_scan_devices(enumerate);
82  struct udev_list_entry *list = udev_enumerate_get_list_entry(enumerate);
83  struct udev_list_entry *entry;
84  udev_list_entry_foreach(entry, list) {
85  const char *path;
86  struct udev_device *dev;
87 
88  // Have to grab the actual udev device here...
89  path = udev_list_entry_get_name(entry);
90  dev = udev_device_new_from_syspath(ud, path);
91 
92  returnVal.append(portInfoFromDevice(dev));
93 
94  // Done with this device
95  udev_device_unref(dev);
96  }
97  // Done with the list and this udev
98  udev_enumerate_unref(enumerate);
99  udev_unref(ud);
100 
101  return returnVal;
102 }

104 bool QextSerialEnumeratorPrivate::setUpNotifications_sys(bool setup)
105 {
106  Q_UNUSED(setup);
107  Q_Q(QextSerialEnumerator);
108  if (!udev) {
109  qCritical() << "Unable to initialize notifications because udev is not initialized.";
110  return false;
111  }
112 
113  // Emit signals immediately for devices already connected (Windows version seems to behave
114  // this way)
115  foreach (QextPortInfo i, getPorts_sys())
116  Q_EMIT q->deviceDiscovered(i);
117 
118  // Look for tty devices from udev.
119  monitor = udev_monitor_new_from_netlink(udev, "udev");
120  udev_monitor_filter_add_match_subsystem_devtype(monitor, "tty", NULL);
121  udev_monitor_enable_receiving(monitor);
122  notifierFd = udev_monitor_get_fd(monitor);
123  notifier = new QSocketNotifier(notifierFd, QSocketNotifier::Read);
124  q->connect(notifier, SIGNAL(activated(int)), q, SLOT(_q_deviceEvent()));
125  notifier->setEnabled(true);
126 
127  return true;
128 }

130 void QextSerialEnumeratorPrivate::_q_deviceEvent()
131 {
132  Q_Q(QextSerialEnumerator);
133  struct udev_device *dev = udev_monitor_receive_device(monitor);
134  if (dev) {
135  QextPortInfo pi = portInfoFromDevice(dev);
136 
137  QLatin1String action(udev_device_get_action(dev));
138  udev_device_unref(dev);
139 
140  if (action == QLatin1String("add"))
141  Q_EMIT q->deviceDiscovered(pi);
142  else if (action == QLatin1String("remove"))
143  Q_EMIT q->deviceRemoved(pi);
144  }
