252  QString sCommPortStatus = MSG_SEARCHING;
253  QString sCommPortDetailedStatus = MSG_SEARCHING;
254  QString sPortName;
255  // Load the current enumerated available ports
256  *pPorts = pEnumerator->getPorts();
257 
258  if(m_serialDevice){
259  // We've previously located the printer, are we still connected?
260  for (int i = 0; i < pPorts->size(); i++) {
261  // Check each existing port to see if our's still exists
262  #ifdef Q_OS_LINUX
263  if(pPorts->at(i).physName == m_serialDevice->portName()){
264  #else
265  if(pPorts->at(i).portName == m_serialDevice->portName()){
266  #endif
267  // We're still connected, set a timer to check again in 5 seconds and then exit
268  QTimer::singleShot(5000, this, SLOT(RefreshCommPortItems()));
269  return;
270  }
271  }
272  // We lost the previous connection and should delete this port connection
273  if (m_serialDevice->isOpen())
274  m_serialDevice->close();
275  delete m_serialDevice;
276  m_serialDevice = NULL;
277  m_Status.reset();
278  qDebug() << sCommPortStatus;
279  emit updateConnectionStatus(sCommPortStatus);
280  sCommPortDetailedStatus = "Lost Comm on previous port. Searching...";
281  emit BC_ConnectionStatusDetailed("Lost Comm on previous port. Searching...");
282  handleLostComm();
283  }
284 
285  // Now we search for a B9Creator
286  int eTime = 1000; // 1 second search cylce time for while not connected
287 
288  sNoFirmwareAurdinoPort = ""; // Reset to null string before scanning ports
289  if(pPorts->size()>0){
290  // Some ports are available, are they the B9Creator?
291  qDebug() << "Scanning For Serial Port Devices (" << pPorts->size() << "found )";
292  for (int i = 0; i < pPorts->size(); i++) {
293  //COMMENTED BECAUSE ITS ANNOYYING...
294  //qDebug() << " port name " << pPorts->at(i).portName;
295  //qDebug() << " locationInfo" << pPorts->at(i).physName;
296  #ifndef Q_OS_LINUX
297  //Note: We only trust friendName, vendorID and productID with Windows and OS_X
298  //COMMENTED BECAUSE ITS ANNOYYING...
299  qDebug() << " description " << pPorts->at(i).friendName;
300  qDebug() << " vendorID " << pPorts->at(i).vendorID;
301  qDebug() << " productID " << pPorts->at(i).productID;
302  #endif
303  #ifdef Q_OS_LINUX
304  // linux ID's ports by physName
305  sPortName = pPorts->at(i).physName;
306  // We filter ports by requiring the portName to begin with "ttyA"
307  if(pPorts->at(i).portName.left(4) == "ttyA" && OpenB9CreatorCommPort(sPortName)){
308  #else
309  // Windows and OSX use portName to ID ports
310  sPortName = pPorts->at(i).portName;
311 
312 
313  // We filter ports by requiring vendorID value of 9025 (Arduino)
314  if(pPorts->at(i).vendorID==9025 && OpenB9CreatorCommPort(sPortName)){
315  #endif
316  // Connected!
317  sCommPortStatus = MSG_CONNECTED;
318  sCommPortDetailedStatus = "Connected on Port: "+m_serialDevice->portName();
319  eTime = 5000; // Re-check connection again in 5 seconds
320  if(m_serialDevice && m_Status.isCurrentVersion())startWatchDogTimer(); // Start B9Creator "crash" watchDog
321  break;
322  }
323  }
324  bool bUpdateFirmware = false;
325  if( m_serialDevice==NULL && sNoFirmwareAurdinoPort!=""){
326  // We did not find a B9Creator with valid firmware, but we did find an Arduino
327  // We assume this is a new B9Creator and needs firmware!
328  // However, we will not upload firmware unless m_bCloneBlanks is true!
329  if(m_bCloneBlanks){
330  // Ask before we clone...
331  QMessageBox vWarn(QMessageBox::Warning,"Upload Firmware?", "A new B9Creator appears to be connected. If this is a newly assembled B9Creator and no other Arduino devices are in use then you should upload the firmware.\n\nProcced with Firmware upload?",QMessageBox::Yes|QMessageBox::No);
332  vWarn.setDefaultButton(QMessageBox::No);

334  bUpdateFirmware = false;
335  }
336  else
337  {
338  //Clone it...
339  qDebug() << "\"Clone Firmware\" Option is enabled. Attempting Firmware Upload to possible B9Creator found on port: "<< sNoFirmwareAurdinoPort;
340  bUpdateFirmware = true;
341  sPortName = sNoFirmwareAurdinoPort;
342  }
343  m_bCloneBlanks = false;
344  }
345  else {
346  qDebug() << "\"Clone Firmware\" Option is disabled. No Firmware upload attempted on possible B9Creator found on port: " << sNoFirmwareAurdinoPort;
347  bUpdateFirmware = false;
348  }
349  }
350  else if (m_serialDevice!=NULL && !m_Status.isCurrentVersion()){
351  // We found a B9Creator with the wrong firmware version, update it!
352  qDebug() << "Incorrect Firmware version found on connected B9Creator"<< sPortName << " Attempting B9Creator Firmware Update";
353  bUpdateFirmware = true;
354  if(m_serialDevice!=NULL) {
355  m_serialDevice->flush();
356  m_serialDevice->close();
357 
358  delete m_serialDevice;
359 
360  }
361  m_bCloneBlanks = false;
362  m_serialDevice = NULL;
363  m_Status.reset();
364  }
365  if(bUpdateFirmware){

367  emit updateConnectionStatus(MSG_FIRMUPDATE);

369  B9FirmwareUpdate Firmware;

371  Firmware.UploadHex(sPortName);
372  QApplication::restoreOverrideCursor();
373  emit updateConnectionStatus(MSG_SEARCHING);
374  emit BC_ConnectionStatusDetailed("Firmware Update Complete. Searching...");
375  }
376  emit updateConnectionStatus(sCommPortStatus);
377  emit BC_ConnectionStatusDetailed(sCommPortDetailedStatus);
378  }
379  QTimer::singleShot(eTime, this, SLOT(RefreshCommPortItems())); // Check again in 5 seconds if connected, 1 secs if not
380 }

382 bool B9PrinterComm::OpenB9CreatorCommPort(QString sPortName)
383 {
384  if(m_serialDevice!=NULL) qFatal("Error: We found an open port handle that should have been deleted!");
385 
386 
387 
388 
389  // Attempt to establish a serial connection with the B9Creator
390  m_serialDevice = new QextSerialPort(sPortName, QextSerialPort::EventDriven, this);
391  if (m_serialDevice->open(QIODevice::ReadWrite) == true) {
392  m_serialDevice->setBaudRate(BAUD115200);
393  m_serialDevice->setDataBits(DATA_8);
394  m_serialDevice->setParity(PAR_NONE);
395  m_serialDevice->setStopBits(STOP_1);
396  m_serialDevice->setFlowControl(FLOW_OFF);
397  m_serialDevice->setDtr(true); // Reset the Aurduino
398  m_serialDevice->setDtr(false);
399 
400  connect(m_serialDevice, SIGNAL(readyRead()), this, SLOT(ReadAvailable()));
401  qDebug() << "Opened Comm port:" << sPortName;
402  }
403  else {
404  // device failed to open
405  if(m_serialDevice!=NULL) delete m_serialDevice;
406  m_serialDevice = NULL;
407  m_Status.reset();
408  qDebug() << "Failed to open Comm port:" << sPortName;
409  return false;
410  }
411 
412  // Delay for up to 5 seconds while we wait for response from printer
413  QTime delayTime = QTime::currentTime().addSecs(5);
414  while( QTime::currentTime() < delayTime && !m_Status.isValidVersion() )
415  QCoreApplication::processEvents(QEventLoop::AllEvents, 100);
416 
417  if(!m_Status.isValidVersion()){
418  if(m_serialDevice!=NULL) {
419  m_serialDevice->flush();
420  m_serialDevice->close();
421  delete m_serialDevice;
422  }
423  m_serialDevice = NULL;
424  m_Status.reset();
425  qDebug() << "Found a possible Arduino Device, perhaps a B9Creator without firmware loaded, on Port: " << sPortName;
426  sNoFirmwareAurdinoPort = sPortName;
427  return false;
428  }
429  return true;

436 void B9PrinterComm::ReadAvailable() {
437  if(m_serialDevice==NULL) qFatal("Error: slot 'ReadAvailable()' but NULL Port Handle");
438 
439  m_Status.resetLastMsgTime(); //update for watchdog
440 
441  if(m_Status.getHomeStatus() == B9PrinterStatus::HS_SEEKING) {
442  m_Status.setHomeStatus(B9PrinterStatus::HS_UNKNOWN); // if we are receiving data, we must no longer be seeking Home
443  // we'll set the status to HS_FOUND once we recieve a 'X' diff broadcast
444  }
445 
446  QByteArray ba = m_serialDevice->readAll(); // read block of available raw data
447 
448  // We process the raw data one line at a time, keeping in mind they may be spread across multiple blocks.
449  int iCurPos = 0;
450  int iLampHrs = -1;
451  int iInput = -1;
452  char c;
453  while(iCurPos<ba.size()){
454  c = ba.at(iCurPos);
455  iCurPos++;
456  if(c!='\r') m_sSerialString+=QString(c);
457  if(c=='\n'){
458  // Line Read Complete, process data
459 
460  if(m_sSerialString.left(1) != "P" && m_sSerialString.left(1) != "L" && m_sSerialString.length()>0){
461  // We only emit this data for display & log purposes
462  if(m_sSerialString.left(1) == "C"){
463  qDebug() << m_sSerialString.right(m_sSerialString.length()-1) << "\n";
464  }
465  else{
466  emit BC_RawData(m_sSerialString);
467  qDebug() << m_sSerialString << "\n";
468  }
469  }
470 
471  int iCmdID = m_sSerialString.left(1).toUpper().toAscii().at(0);
472  QMessageBox vWarn(QMessageBox::Critical,"Runaway X Motor!","X Motor Failure Detected, Code: " + m_sSerialString +"\nClose program and troubleshoot hardware.");
473  switch (iCmdID){
474  case 'U': // Mechanical failure of encoder?
475  SendCmd("S"); // Stop motions
476  SendCmd("P0");// Turn off projector
477  qDebug() << "WARNING: Printer has sent 'U' report, runaway X Motor indication: " + m_sSerialString << "\n";
478  vWarn.exec();
479  break;
480  case 'Q': // Printer got tired of waiting for command and shut down projectors
481  // We will likely never see this as something bad has happened
482  // (like we have crashed or been shut off during a print process
483  // So if we get it, we'll just post it to the log and wait for
484  // timeouts to correct things.
485  qDebug() << "WARNING: Printer has sent 'Q' report, lost comm with host." << "\n";
486  break;
487  case 'P': // take care of projector status
488  iInput = m_sSerialString.right(m_sSerialString.length()-1).toInt();
489  if(iInput!=1)iInput = 0;
490  if(handleProjectorBC(iInput)){
491  //projector status changed
492  emit BC_RawData(m_sSerialString);
493  qDebug() << m_sSerialString << "\n";
494  }
495  break;
496  case 'L': // take care of projector Lamp hours update
497  iLampHrs = m_sSerialString.right(m_sSerialString.length()-1).toInt();
498  if(m_Status.getLampHrs()!= iLampHrs){
499  m_Status.setLampHrs(iLampHrs);
500  emit BC_ProjectorStatusChanged();
501  emit BC_RawData(m_sSerialString);
502  qDebug() << m_sSerialString << "\n";
503  }
504  break;
505 
506  case 'X': // Found Home with this Difference Offset
507  m_Status.setHomeStatus(B9PrinterStatus::HS_FOUND);
508  m_Status.setLastHomeDiff(m_sSerialString.right(m_sSerialString.length()-1).toInt());
509  emit BC_HomeFound();
510  break;
511 
512  case 'R': // Needs Reset?
513  iInput = m_sSerialString.right(m_sSerialString.length()-1).toInt();
514  if(iInput==0) m_Status.setHomeStatus(B9PrinterStatus::HS_FOUND);
515  else m_Status.setHomeStatus(B9PrinterStatus::HS_UNKNOWN);
516  break;
517 
518  case 'K': // Current Lamp 1/2 life
519  m_Status.setHalfLife(m_sSerialString.right(m_sSerialString.length()-1).toInt());
520  emit BC_HalfLife(m_Status.getHalfLife());
521  break;
522 
523  case 'D': // Current Native X Projector resolution
524  m_Status.setNativeX(m_sSerialString.right(m_sSerialString.length()-1).toInt());
525  emit BC_NativeX(m_Status.getNativeX());
526  break;
527 
528  case 'E': // Current Native Y Projector resolution
529  m_Status.setNativeY(m_sSerialString.right(m_sSerialString.length()-1).toInt());
530  emit BC_NativeY(m_Status.getNativeY());
531  break;
532 
533  case 'H': // Current XY Pixel Size
534  m_Status.setXYPixelSize(m_sSerialString.right(m_sSerialString.length()-1).toInt());
535  emit BC_XYPixelSize(m_Status.getXYPixelSize());
536  break;
537 
538  case 'I': // Current PU
539  m_Status.setPU(m_sSerialString.right(m_sSerialString.length()-1).toInt());
540  emit BC_PU(m_Status.getPU());
541  break;
542 
543  case 'A': // Projector Control capability
544  m_Status.setProjectorRemoteCapable(m_sSerialString.right(m_sSerialString.length()-1).toInt());
545  emit BC_ProjectorRemoteCapable(m_Status.isProjectorRemoteCapable());
546  break;
547 
548  case 'J': // Projector Shutter capability
549  m_Status.setHasShutter(m_sSerialString.right(m_sSerialString.length()-1).toInt());
550  emit BC_HasShutter(m_Status.hasShutter());
551  break;
552 
553  case 'M': // Current Z Upper Limit in PUs
554  m_Status.setUpperZLimPU(m_sSerialString.right(m_sSerialString.length()-1).toInt());
555  emit BC_UpperZLimPU(m_Status.getUpperZLimPU());
556  break;
557 
558  case 'Z': // Current Z Position Update
559  m_Status.setCurZPosInPU(m_sSerialString.right(m_sSerialString.length()-1).toInt());
560  emit BC_CurrentZPosInPU(m_Status.getCurZPosInPU());
561  break;
562 
563  case 'S': // Current Vat(Shutter) Percent Open Position Update
564  m_Status.setCurVatPercentOpen(m_sSerialString.right(m_sSerialString.length()-1).toInt());
565  emit BC_CurrentVatPercentOpen(m_Status.getCurVatPercentOpen());
566  break;
567 
568  case 'C': // Comment
569  emit BC_Comment(m_sSerialString.right(m_sSerialString.length()-1));
570  break;
571 
572  case 'F': // Print release cycle finished
573  emit BC_PrintReleaseCycleFinished();

613 bool B9PrinterComm::handleProjectorBC(int iBC){
614  bool bStatusChanged = false;
615  if(m_Status.isProjectorPowerCmdOn() && iBC == 0){
616  // Projector commanded ON but current report is OFF
617  switch(m_Status.getProjectorStatus()){
618  case B9PrinterStatus::PS_OFF:
619  case B9PrinterStatus::PS_UNKNOWN:
620  case B9PrinterStatus::PS_TIMEOUT:
621  case B9PrinterStatus::PS_FAIL:
622  setProjectorPowerCmd(true); // turn it on now
623  bStatusChanged = true;
624  break;
625  case B9PrinterStatus::PS_TURNINGON:
626  if(m_Status.getLastProjCmdElapsedTime()>45000){
627  // Taking too long to turn on, something is wrong!
628  m_Status.setProjectorStatus(B9PrinterStatus::PS_TIMEOUT);
629  emit BC_ProjectorFAIL();
630  m_Status.cmdProjectorPowerOn(false);
631  bStatusChanged = true;
632  }
633  break;
634  case B9PrinterStatus::PS_WARMING:
635  case B9PrinterStatus::PS_ON:
636  // Uncommanded Power off! Lost power or bulb failure?
637  m_Status.setProjectorStatus(B9PrinterStatus::PS_FAIL);
638  emit BC_ProjectorFAIL();
639  m_Status.cmdProjectorPowerOn(false);
640  bStatusChanged = true;
641  break;
642  case B9PrinterStatus::PS_COOLING:
643  // Done cooling off
644  m_Status.setProjectorStatus(B9PrinterStatus::PS_OFF);
645  emit BC_ProjectorStatusChanged();
646  bStatusChanged = true;
647  break;
648  default:
649  // Nothing we can do
650  break;
651  }
652  }
653  else if(m_Status.isProjectorPowerCmdOn() && iBC == 1){
654  // Projector commanded ON and current report is ON
655  switch(m_Status.getProjectorStatus()){
656  case B9PrinterStatus::PS_COOLING:
657  case B9PrinterStatus::PS_OFF:
658  case B9PrinterStatus::PS_UNKNOWN:
659  case B9PrinterStatus::PS_TIMEOUT:
660  case B9PrinterStatus::PS_FAIL:
661  // We were turning off, off, failed, timed out or unknown and suddenly we're cmd on and actually on?
662  // Best we can do is set the status on and report
663  m_Status.setProjectorStatus(B9PrinterStatus::PS_ON);
664  emit BC_ProjectorStatusChanged();
665  bStatusChanged = true;
666  break;
667  case B9PrinterStatus::PS_TURNINGON:
668  // We were turning on and now we need to warm up a bit
669  m_Status.setProjectorStatus(B9PrinterStatus::PS_WARMING);
670  startWarmingTime.start();
671  emit BC_ProjectorStatusChanged();
672  bStatusChanged = true;
673  break;
674  case B9PrinterStatus::PS_WARMING:
675  if(startWarmingTime.elapsed()>m_iWarmUpDelayMS){
676  // All warmed up now, ready to use.
677  m_Status.setProjectorStatus(B9PrinterStatus::PS_ON);
678  emit BC_ProjectorStatusChanged();
679  bStatusChanged = true;
680  }
681  break;
682  case B9PrinterStatus::PS_ON:
683  default:
684  // No change, we're good.
685  break;
686  }
687  }
688  else if(!m_Status.isProjectorPowerCmdOn() && iBC == 0){
689  // Projector commanded OFF and current report is OFF
690  switch(m_Status.getProjectorStatus()){
691  case B9PrinterStatus::PS_UNKNOWN:
692  case B9PrinterStatus::PS_TURNINGON:
693  case B9PrinterStatus::PS_WARMING:
694  case B9PrinterStatus::PS_ON:
695  case B9PrinterStatus::PS_COOLING:
696  case B9PrinterStatus::PS_TIMEOUT:
697  case B9PrinterStatus::PS_FAIL:
698  // We were turning on, warming up, on, cooling off, unknown, timed out or failed and now we're off
699  m_Status.setProjectorStatus(B9PrinterStatus::PS_OFF);
700  emit BC_ProjectorStatusChanged();
701  bStatusChanged = true;
702  break;
703  case B9PrinterStatus::PS_OFF:
704  default:
705  // No change, we're good
706  break;
707  }
708  }
709  else if(!m_Status.isProjectorPowerCmdOn() && iBC == 1){
710  // Projector commanded OFF but current report is ON
711  switch(m_Status.getProjectorStatus()){
712  case B9PrinterStatus::PS_COOLING:
713  if(m_Status.getLastProjCmdElapsedTime()>30000){
714  // Taking too long to turn off, send the off command again
715  m_Status.setProjectorStatus(B9PrinterStatus::PS_ON);
716  emit BC_ProjectorStatusChanged();
717  bStatusChanged = true;
718  }
719  break;
720  case B9PrinterStatus::PS_OFF:
721  //User must have switched it on manually, we'll accept that and set the status to commanded on and Turning on
722  cmdProjectorPowerOn(true);
723  setProjectorPowerCmd(true);
724  bStatusChanged = true;
725  break;
726 
727  case B9PrinterStatus::PS_TURNINGON:
728  case B9PrinterStatus::PS_WARMING:
729  case B9PrinterStatus::PS_ON:
730  case B9PrinterStatus::PS_UNKNOWN:
731  case B9PrinterStatus::PS_TIMEOUT:
732  case B9PrinterStatus::PS_FAIL:
733  default:
734  setProjectorPowerCmd(false); // turn it off now
735  bStatusChanged = true;
736  break;
737  }
738  }
739  return bStatusChanged;
740 }
