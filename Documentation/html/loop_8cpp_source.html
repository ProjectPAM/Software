98 //return 0 if failed to complete a loop!
99 {
100  bool grow = 1;
101  Segment* thisSeg = pOriginSeg;
102  Segment* nextSeg = pOriginSeg->leadingSeg;
103 
104  do
105  {
106  thisSeg = segListp[numSegs - 1];//the last one added
107  nextSeg = thisSeg->leadingSeg;
108 
109  if(nextSeg)
110  {
111 
112 
113  if(nextSeg->pLoop)//if the next segment is already part of ANY loop
114  {
115  if(nextSeg->pLoop == this)//is it THIS loop?
116  {
117  if(nextSeg == pOriginSeg)//if we have met up to the BEGINING of our loop
118  {
119  //this is the normal completion option
120  isComplete = true;
121  grow = 0;
122  }
123  else//the loop has crossed itself too soon! should NOT happen, because it is impossible for 2 trailing segment to exist on 1 segment
124  {
125  qDebug() << "Loop Crossover at " << numSegs << " segments, Check Neighbor code!";
126  grow = 0;
127  }
128  }
129  else//or another loop? (branch off)
130  {
131  //also shoudnt happen
132  qDebug() << "De-Rail Detected at " << numSegs << " segments, Check Neighbor code!";
133  grow = 0;
134  }
135  }
136  else//the next segment hasnt been claimed yet.
137  {
138  AttachSegment(nextSeg);
139  grow = 1;//for clarity
140  }
141 
142  }
143  else//dead end!
144  {
145  SealHole(thisSeg);
146  isPatched = true;
147  isComplete = true;

150  }while(grow);

288  double ThisCross;
289  unsigned int s1;
290  unsigned int s2;
291  for(s1=0;s1<segListp.size(); s1++)
292  {

298  {

300  }
301 
302  if(IsZero(Distance2D(segListp[s1]->p2, segListp[s2]->p2),0.001))//we need to nudge
303  {
304  segListp[s1]->p2 = segListp[s2]->p2;
305  segListp[s1]->leadingSeg->p1 = segListp[s1]->p2;
306  segListp[s1]->FormNormal();
307  segListp[s1]->leadingSeg->FormNormal();
308 
309 
310  QVector2D aveVec((segListp[s1]->normal.x() + segListp[s1]->leadingSeg->normal.x())/2.0, (segListp[s1]->normal.y() + segListp[s1]->leadingSeg->normal.y())/2.0);
311  aveVec.normalize();
312 
313  ThisCross = (segListp[s1]->normal.x() * segListp[s1]->leadingSeg->normal.y()) - (segListp[s1]->normal.y() * segListp[s1]->leadingSeg->normal.x());
314 
315  if(fabs(ThisCross) < 0.001)
316  {
317  aveVec = (segListp[s2]->normal + segListp[s2]->leadingSeg->normal)/2.0;
318  segListp[s1]->p2 += (aveVec*0.002);
319  }
320  else
321  {
322  if(ThisCross > 0)
323  {
324  segListp[s1]->p2 += (aveVec*0.002);
325  }
326  else
327  {
328  segListp[s1]->p2 -= (aveVec*0.002);
329  }
330  }
331 
332  segListp[s1]->leadingSeg->p1 = segListp[s1]->p2;
333 
334  segListp[s1]->FormNormal();
335  segListp[s1]->leadingSeg->FormNormal();
336  nudges++;
337  }
338  }
339  }
