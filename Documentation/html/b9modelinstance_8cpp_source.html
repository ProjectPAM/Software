631 void B9ModelInstance::ScaleSupportPositionsAroundCenter(QVector3D newScale, QVector3D oldScale)
632 {
633  //"Scaling" the supports is really just changing
634  //their positions accourdingly.
635  // were given a delta scale which is the different between
636  //the old scale vector and the new scale vector.
637  unsigned int s;
638  B9SupportStructure* cSprt;
639  QVector3D oldTopLocalPosition;
640  QVector3D oldBottomLocalPosition;
641  QVector3D newTopLocalPosition;
642  QVector3D newBottomLocalPosition;
643 
644  QVector3D moveFactor;
645  moveFactor = oldScale*QVector3D(1/newScale.x(),1/newScale.y(),1/newScale.z());
646 
647  for(s = 0; s < supportStructureList.size(); s++)
648  {
649  cSprt = supportStructureList[s];
650 
651  oldTopLocalPosition = cSprt->GetTopPoint();
652  oldBottomLocalPosition = cSprt->GetBottomPoint();
653 
654  //move old positions into local model rotation domain.
655  RotateVector(oldTopLocalPosition,-this->rot.z(),QVector3D(0,0,1));
656  RotateVector(oldTopLocalPosition,-this->rot.x(),QVector3D(1,0,0));
657  RotateVector(oldTopLocalPosition,-this->rot.y(),QVector3D(0,1,0));
658  RotateVector(oldBottomLocalPosition,-this->rot.z(),QVector3D(0,0,1));
659  RotateVector(oldBottomLocalPosition,-this->rot.x(),QVector3D(1,0,0));
660  RotateVector(oldBottomLocalPosition,-this->rot.y(),QVector3D(0,1,0));
661 
662 
663  //do the resulting translation
664 
665  newTopLocalPosition = oldTopLocalPosition*moveFactor;
666  newBottomLocalPosition = oldBottomLocalPosition*moveFactor;
667 
668  //go back to global domain.
669  RotateVector(newTopLocalPosition,this->rot.y(),QVector3D(0,1,0));
670  RotateVector(newTopLocalPosition,this->rot.x(),QVector3D(1,0,0));
671  RotateVector(newTopLocalPosition,this->rot.z(),QVector3D(0,0,1));
672  RotateVector(newBottomLocalPosition,this->rot.y(),QVector3D(0,1,0));
673  RotateVector(newBottomLocalPosition,this->rot.x(),QVector3D(1,0,0));
674  RotateVector(newBottomLocalPosition,this->rot.z(),QVector3D(0,0,1));
675 
676  cSprt->SetTopPoint(newTopLocalPosition);
677  cSprt->SetBottomPoint(newBottomLocalPosition);
678 
679  }
680 
681  //additionally we want to deal with the basePlate and scale the radius
682  if(GetBasePlate())
683  {
684  GetBasePlate()->SetBottomRadius(GetBasePlate()->GetBottomRadius()*moveFactor.x());
685  }

690 void B9ModelInstance::FlipSupports()
691 {
692  unsigned int s;
693  B9SupportStructure* cSprt;
694  QVector3D oldTopLocalPosition;
695  QVector3D oldBottomLocalPosition;
696  QVector3D newTopLocalPosition;
697  QVector3D newBottomLocalPosition;
698  QVector3D TopNorm;
699  QVector3D BottomNorm;
700 
701 
702  QVector3D mirrorVec(0,1,0);
703  RotateVector(mirrorVec,this->rot.z(),QVector3D(0,0,1));
704  QVector3D flipVec(0,1,0);
705  RotateVector(flipVec,this->rot.z()-90,QVector3D(0,0,1));
706 
707  double side;
708  double distToPlane;
709 
710  for(s = 0; s < supportStructureList.size(); s++)
711  {
712  cSprt = supportStructureList[s];
713 
714  //top
715  oldTopLocalPosition = cSprt->GetTopPoint();
716  distToPlane = QVector3D().distanceToLine(QVector2D(oldTopLocalPosition),mirrorVec);
717  side = QVector3D::crossProduct(mirrorVec,QVector2D(oldTopLocalPosition).normalized()).z();
718  if(side < 0) side = -1.0;
719  else side = 1.0;
720  newTopLocalPosition = oldTopLocalPosition + side*2.0*distToPlane*flipVec;
721 
722  //bottom
723  oldBottomLocalPosition = cSprt->GetBottomPoint();
724  distToPlane = QVector3D().distanceToLine(QVector2D(oldBottomLocalPosition),mirrorVec);
725  side = QVector3D::crossProduct(mirrorVec,QVector2D(oldBottomLocalPosition).normalized()).z();
726  if(side < 0) side = -1.0;
727  else side = 1.0;
728  newBottomLocalPosition = oldBottomLocalPosition + side*2.0*distToPlane*flipVec;
729 
730 
731  //topnorm
732  TopNorm = cSprt->GetTopNormal();
733  distToPlane = QVector3D().distanceToLine(QVector2D(TopNorm),mirrorVec);
734  side = QVector3D::crossProduct(mirrorVec,QVector2D(TopNorm).normalized()).z();
735  if(side < 0) side = -1.0;
736  else side = 1.0;
737  TopNorm = TopNorm + side*distToPlane*2.0*flipVec;
738  TopNorm.normalize();
739 
740 
741  //bottomnorm
742  BottomNorm = cSprt->GetBottomNormal();
743  distToPlane = QVector3D().distanceToLine(QVector2D(BottomNorm),mirrorVec);
744  side = QVector3D::crossProduct(mirrorVec,QVector2D(BottomNorm).normalized()).z();
745  if(side < 0) side = -1.0;
746  else side = 1.0;
747  BottomNorm = BottomNorm + side*distToPlane*2.0*flipVec;
