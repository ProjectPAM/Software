42 void floodFill(QImage* pImage, int x, int y, QColor fillColor)
43 {
44  // If the pixel we are starting with is already the fill color, we're done
45  if (pImage->pixel(x,y) == fillColor.rgb()) return;
46 
47  int startcolor = pImage->pixel(x,y);
48 
49  // Create the pixel queue. Assume the worst case where every pixel in the
50  // image may be in the queue.
51  int pixelQueueSize = 0;
52  int *pixelQueue= new int[pImage->width() * pImage->height()];
53 
54  // Add the start pixel to the queue (we created a single array of ints,
55  // even though we are enqueuing two numbers. We put the y value in the
56  // upper 16 bits of the integer, and the x in the lower 16. This gives
57  // a limit of 65536x65536 pixels, that should be enough.)
58  pixelQueue[0] = (y << 16) + x;
59  pixelQueueSize = 1;
60 
61  // Color the start pixel.
62  pImage->setPixel(x,y,fillColor.rgb());
63 
64  // Keep going while there are pixels in the queue.
65  while (pixelQueueSize > 0){
66 
67  // Get the x and y values of the next pixel in the queue
68  x = pixelQueue[0] & 0xffff;
69  y = (pixelQueue[0] >> 16) & 0xffff;
70 
71  // Remove the first pixel from the queue. Rather than move all the
72  // pixels in the queue, which would take forever, just take the one
73  // off the end and move it to the beginning (order doesn't matter here).
74  pixelQueueSize--;
75  pixelQueue[0] = pixelQueue[pixelQueueSize];
76 
77  // If we aren't on the left side of the image, see if the pixel to the
78  // left has been painted. If not, paint it and add it to the queue.
79  if (x > 0) {
80  if ( pImage->pixel(x-1, y) == startcolor )
81  {
82  pImage->setPixel(x-1, y, fillColor.rgb());
83  pixelQueue[pixelQueueSize] = (y << 16) + x-1;
84  pixelQueueSize++;
85  }
86  }
87 
88  // If we aren't on the bottom of the image, see if the pixel below
89  // this one has been painted. If not, paint it and add it to the queue.
90  if (y > 0) {
91  if ( pImage->pixel(x, y-1) == startcolor )
92  {
93  pImage->setPixel(x, y-1, fillColor.rgb());
94  pixelQueue[pixelQueueSize] = ((y-1) << 16) + x;
95  pixelQueueSize++;
96  }
97  }
98 
99  // If we aren't on the right side of the image, see if the pixel to the
100  // right has been painted. If not, paint it and add it to the queue.
101  if (x < pImage->width()-1) {
102  if ( pImage->pixel(x+1, y) == startcolor )
103  {
104  pImage->setPixel(x+1, y, fillColor.rgb());
105  pixelQueue[pixelQueueSize] = (y << 16) + x+1;
106  pixelQueueSize++;
107  }
108  }
109 
110  // If we aren't on the top of the image, see if the pixel above
111  // this one has been painted. If not, paint it and add it to the queue.
112  if (y < pImage->height()-1) {
113  if ( pImage->pixel(x, y+1) == startcolor )
114  {
115  pImage->setPixel(x, y+1, fillColor.rgb());
116  pixelQueue[pixelQueueSize] = ((y+1) << 16) + x;
117  pixelQueueSize++;
118  }
119  }
120  }
121  delete [] pixelQueue;
122 }
