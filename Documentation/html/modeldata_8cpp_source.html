205 void ModelData::CenterModel()
206 {
207  //figure out what to current center of the models counds is..
208  unsigned int f;
209  unsigned int v;
210  QVector3D center;
211 
212 
213  for(f=0;f<triList.size();f++)
214  {
215  for(v=0;v<3;v++)
216  {
217  //update the models bounds.
218  //max
219  if(triList[f].maxBound.x() > maxbound.x())
220  {
221  maxbound.setX(triList[f].vertex[v].x());
222  }
223  if(triList[f].maxBound.y() > maxbound.y())
224  {
225  maxbound.setY(triList[f].vertex[v].y());
226  }
227  if(triList[f].maxBound.z() > maxbound.z())
228  {
229  maxbound.setZ(triList[f].vertex[v].z());
230  }
231 
232  //mins
233  if(triList[f].minBound.x() < minbound.x())
234  {
235  minbound.setX(triList[f].vertex[v].x());
236  }
237  if(triList[f].minBound.y() < minbound.y())
238  {
239  minbound.setY(triList[f].vertex[v].y());
240  }
241  if(triList[f].minBound.z() < minbound.z())
242  {
243  minbound.setZ(triList[f].vertex[v].z());
244  }
245  }
246  }
247 
248  center = (maxbound + minbound)*0.5;
249 
250  for(f=0;f<triList.size();f++)
251  {
252  for(v=0;v<3;v++)
253  {
254  triList[f].vertex[v] -= center;
255  }
256  triList[f].UpdateBounds(); // since we are moving every triangle, we need to update their bounds too.
257  }
258  maxbound -= center;
259  minbound -= center;
260 }
