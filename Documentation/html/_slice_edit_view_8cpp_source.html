162 void SliceEditView::GoToSlice(int slicenumber)
163 {
164  //make sure the suggested slice is in range...
165  if(slicenumber > pCPJ->getTotalLayers() - 1 || slicenumber < 0)
166  {
167  return;
168  }
169  if(slicenumber != currSlice)
170  {
171  ClearUndoBuffer();
172  }
173 
174  currSlice = slicenumber;
175  setWindowTitle("Slice Manager - " + GetEditMode() + ": " + QString().number(currSlice+1) + " / " + QString().number(pCPJ->getTotalLayers()));
176  DeCompressIntoContext();
177 
178  if(supportMode)
179  {
180  RefreshContext(0);
181  }
182  else
183  {
184  RefreshContext(1);
185  }
186 
187  if(!supportMode)
188  {
189  SaveToUndoBuffer(); //when moving on to a new slice, always save the state it's in to a buffer.
190  }
191 
192  greenTimer.start();
193 }
194 //inflates the image from the job file, and also inflates the one below if in support mode.
195 void SliceEditView::DeCompressIntoContext()
196 {
197  pCPJ->setCurrentSlice(currSlice);
198  if(supportMode)
199  pCPJ->showSupports(1);
200  else
201  pCPJ->showSupports(0);
202 
203  //double check if the top image is 0x0, we want to resize it to fix the "grey window"
204  //problem when first opening an image.
205  //because
206  if(topImg.size().width() == 0)
207  {
208  topImg = QImage(1024,768,QImage::Format_ARGB32_Premultiplied) ;
209  }
210 
211 
212 
213  pCPJ->inflateCurrentSlice(&topImg, m_xOffset, m_yOffset, true);
214  pDrawingContext->SetUpperImg(&topImg);
215 
216  if(currSlice <= 0)//make "base" image
217  {
218  botImg = QImage(topImg.width(),topImg.height(),QImage::Format_RGB16);
219  botImg.fill(QColor(255,255,255));
220  }
221  else
222  {
223  pCPJ->setCurrentSlice(currSlice - 1);
224  pCPJ->inflateCurrentSlice(&botImg, m_xOffset, m_yOffset, true);
225  }
226 
227  pDrawingContext->SetLowerImg(&botImg);
228 }

321 void SliceEditView::SaveToUndoBuffer()
322 {
323  int i;
324  for(i = imgBackup.size() - 1; i > backupIndx; i--)
325  {
326  imgBackup.removeAt(i);
327  }
328  //add to the list of undos.

330  if(imgBackup.size() > 25)
331  {imgBackup.removeFirst();}
332  backupIndx = imgBackup.size() - 1;

334 void SliceEditView::Undo()
335 {
336  if(!supportMode)
337  {
338  if(backupIndx >= 1)
339  {
340  backupIndx--;
341  topImg = imgBackup[backupIndx];
342  pDrawingContext->SetUpperImg(&topImg);
343  RefreshContext(1);
344  modified = true;
345  ReCompress();
346  pDrawingContext->update();
347  }
348  }

350 void SliceEditView::Redo()
351 {
352  if(!supportMode)
353  {
354  backupIndx++;
355  if(backupIndx >= imgBackup.size())
356  {
357  backupIndx = imgBackup.size() - 1;
358  return;
359  }
360 
361  topImg = imgBackup[backupIndx];
362  pDrawingContext->SetUpperImg(&topImg);
363  RefreshContext(1);
364  modified = true;
365  ReCompress();
366  pDrawingContext->update();
367  }
