479 void B9Projector::createNormalizedMask( double XYPS, double dZ, double dOhMM)
480 {
481  if(mpCPJ!=NULL)
482  XYPS = mpCPJ->getXYPixelmm();
483  double xsqrmm, ysqrmm;
484  double zsqrmm = dZ*dZ;
485  double Oh = dOhMM;
486  double Ol = width()*XYPS*0.5;
487  double dDimRg = 255.0/sqrt(Ol*Ol + Oh*Oh + zsqrmm);
488 
489  //Here we create a gray scale mask to normalize the projector's output
490  m_NormalizedMask = QImage(width(),height(),QImage::Format_ARGB32_Premultiplied);
491  m_NormalizedMask.fill(qRgba(0,0,0,0));
492  int igs;
493  for (quint32 y = 0; y < (quint32)m_NormalizedMask.height(); ++y) {
494  QRgb *scanLine = (QRgb *)m_NormalizedMask.scanLine(y);
495  for (quint32 x = 0; x < (quint32)m_NormalizedMask.width(); ++x){
496  xsqrmm = pow((-Ol + (double)x*0.1),2.0);
497  ysqrmm = pow(( Oh - (double)y*0.1),2.0);
498  igs = sqrt(xsqrmm+ysqrmm+zsqrmm)*dDimRg;
499  scanLine[x] = qRgba(igs,igs,igs,255);
500  }
501  }
502 }
