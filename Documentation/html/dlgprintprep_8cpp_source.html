177  int iM = iTime/60000;
178  int iH = iM/60;
179  iM = (int)((double)iM+0.5) - iH*60;
180  QString sLZ = ":0"; if(iM>9)sLZ = ":";
181  QString sTimeRemaining = QString::number(iH)+sLZ+QString::number(iM);
182  t.setHMS(0,0,0); vTimeRemains = t.addMSecs(iTime);
183  ui->lcdNumberTimeRequired->display(vTimeRemains.toString("hh:mm"));
184 
185  double dVolume = m_pCPJ->getTotalWhitePixels(0,m_iLastLayer)*m_pCPJ->getZLayermm()*m_pCPJ->getXYPixelmm()*m_pCPJ->getXYPixelmm()/1000;
186  ui->lineEditVolume->setText(QString::number(dVolume,'f',1));
