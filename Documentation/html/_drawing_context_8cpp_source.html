84 void DrawingContext::GenerateLogicImage()
85 {

89  int x;

92  int grey = QColor(190,190,190).rgb();
93  int black = QColor(0,0,0).rgb();

95  int width = pActiveImage->width();
96  int height = pActiveImage->height();
97 
98 
99 
100  for(x=0;x<width;x++)
101  {
102  for(y=0;y<height;y++)
103  {
104  if(pActiveImage->pixel(x,y) == white || pActiveImage->pixel(x,y) == red || pActiveImage->pixel(x,y) == grey)
105  {
106  if(pLowerImage->pixel(x,y) == black)
107  {
108  pActiveImage->setPixel(x,y,red);
109  }
110  else

113  }
114  }

175 void DrawingContext::mousePressEvent(QMouseEvent *event)
176  {
177  if (event->button() == Qt::LeftButton) {
178  if(pActiveImage)
179  {
180  if(pEditView->currSlice < pEditView->pCPJ->getBase())
181  {
182  QMessageBox::StandardButton ret;
183  ret = QMessageBox::warning(this, tr("Slice Manager"),
184  tr("Can't Edit Base Slices"), QMessageBox::Ok);
185  return;
186  }
187  if(pEditView->supportMode)
188  {
189  if(deletemode)
190  {
191  //delete supports
192  if(pEditView->pCPJ->DeleteSupport(pEditView->currSlice, event->pos(), 16))
193  {
194  pEditView->DeCompressIntoContext();
195  GenerateLogicImage();
196  pEditView->RefreshWithGreen();
197  }
198 
199  }
200  else
201  {
202  if(currSupportTool == "circle")
203  {
204  pEditView->AddSupport(event->pos(),supportSize,st_CIRCLE,fastsupportmode);
205  }
206  else if(currSupportTool == "square")
207  {
208  pEditView->AddSupport(event->pos(),supportSize,st_SQUARE,fastsupportmode);
209  }
210  else if(currSupportTool == "triangle")
211  {
212  pEditView->AddSupport(event->pos(),supportSize,st_TRIANGLE,fastsupportmode);
213  }
214  else if(currSupportTool == "diamond")
215  {
216  pEditView->AddSupport(event->pos(),supportSize,st_DIAMOND,fastsupportmode);
217  }
218  }
219  }
220  else
221  {
222  scribbling = true;
223 
224  if(currDrawTool == "penfill" || currDrawTool == "penvoid")
225  {
226  lastPoint = event->pos();
227  }
228  else if(currDrawTool == "floodfill" || currDrawTool == "floodvoid")
229  {
230  floodFill(pActiveImage, event->pos().x(), event->pos().y(), PenColor);
231  pEditView->modified = true;
232  }
233  }
234 
235  }
236  }
237  }
