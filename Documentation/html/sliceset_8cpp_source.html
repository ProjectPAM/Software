114 Slice* SliceSet::ParallelCreateSlices(bool &slicesInTransit, CrushedPrintJob* toJob)
115 {
116  unsigned int i;
117  Slice* pSlice = NULL;
118  SliceRequest req;
119  int buzythreads = 0;
120  SetupFutureWorkers(); // make sure our thread count matches our cores (recomended core count)
121  SetupRasturizer();
122 
123  //Deploy workers to smart jobs!
124  for(i = 0; i < workerThreads.size(); i++)
125  {
126  if(workerThreads[i].isRunning()){
127  buzythreads++;
128  continue;
129  }
130 
131  //COMPRESSION
132  if(RasturizedSlices.size() && toJob)
133  {
134 
135  pSlice = RasturizedSlices.front();
136  if(!pSlice->inProccessing)
137  {
138  RasturizedSlices.pop();
139 
140  pSlice->inProccessing = true;
141  CompressedSlices.push(pSlice);
142  workerThreads[i] = QtConcurrent::run(this, &SliceSet::AddSliceToJob, pSlice, toJob);
143 
144  continue;
145  }
146  }
147 
148  //SLICING
149  if(SliceRequests.size() && SlicedSlices.size() < 20)//19 slices max in buffer..
150  {
151  req = SliceRequests.front();
152  SliceRequests.pop();
153 
154  pSlice = new Slice(req.altitude,req.layerIndx);
155  pSlice->inProccessing = true;
156  SlicedSlices.push(pSlice);
157  workerThreads[i] = QtConcurrent::run(this, &SliceSet::ComputeSlice, pSlice);
158  continue;
159  }
160 
161 
162  }
163  //rasturizing
164  if(SlicedSlices.size() && toJob)
165  {
166  pSlice = SlicedSlices.front();
167  if(!pSlice->inProccessing)
168  {
169  SlicedSlices.pop();
170  pSlice->inProccessing = true;
171  RasterizeSlice(pSlice);
172  RasturizedSlices.push(pSlice);
173  }
174  }
175 
176 
177  if(toJob)
178  {
179  if(SliceRequests.size() || SlicedSlices.size() || RasturizedSlices.size() || CompressedSlices.size())
180  slicesInTransit = true;
181  else
182  slicesInTransit = false;
183  if(CompressedSlices.size() && !CompressedSlices.front()->inProccessing){
184  pSlice = CompressedSlices.front();
185  CompressedSlices.pop();
186  return pSlice;}
187  else
188  return NULL;
189  }
190  else
191  {
192  if(SliceRequests.size() || SlicedSlices.size())
193  slicesInTransit = true;
194  else
195  slicesInTransit = false;
196 
197  if(SlicedSlices.size() && !SlicedSlices.front()->inProccessing){
198  pSlice = SlicedSlices.front();
199  SlicedSlices.pop();
200  return pSlice;}
201  else
202  return NULL;
203  }

241

283  QRgb pickedcolor;

294  if(result > 0)
295  {
296  result = 255;
297  }else result = 0;
298  img->setPixel(x,y,QColor(result,0,0).rgb());
299  }

301  }
