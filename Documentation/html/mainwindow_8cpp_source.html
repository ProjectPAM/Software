307 void MainWindow::AttemptPrintDialogWithFile(QString openFile)
308 {
310  // Open the .b9j file
311  m_pCPJ->clearAll();
312 
313  QFile file(openFile);
314  if(!m_pCPJ->loadCPJ(&file)) {
315  QMessageBox msgBox;
316  msgBox.setText("Error Loading File. Unknown Version?");
317  msgBox.exec();
318  return;
319  }
320  m_pCPJ->showSupports(true);
321  int iXYPixelMicrons = m_pCPJ->getXYPixelmm()*1000;
322  if( pTerminal->isConnected()&& iXYPixelMicrons != (int)pTerminal->getXYPixelSize()){
323  QMessageBox msgBox;
324  msgBox.setText("WARNING");
325  msgBox.setInformativeText("The XY pixel size of the selected job file ("+QString::number(iXYPixelMicrons)+" m) does not agree with the Printer's calibrated XY pixel size ("+QString::number(pTerminal->getXYPixelSize())+" m)!\n\n"
326  "Printing will likely result in an object with incorrect scale and/or apsect ratio.\n\n"
327  "Do you wish to continue?");
328  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
329  msgBox.setDefaultButton(QMessageBox::No);
330  int ret = msgBox.exec();
331  if(ret==QMessageBox::No)return;
332  }
333 
334 
335  m_pPrintPrep = new DlgPrintPrep(m_pCPJ, pTerminal, this);
336  connect (m_pPrintPrep, SIGNAL(accepted()),this,SLOT(doPrint()));
337  m_pPrintPrep->exec();
338 }
