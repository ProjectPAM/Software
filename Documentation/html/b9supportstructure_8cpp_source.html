67 void B9SupportAttachmentData::CenterGeometry()
68 {
69  unsigned int t,v;
70  Triangle3D* cTri = NULL;
71  QVector3D MinBound(999999,999999,999999);
72  QVector3D MaxBound(-999999,-9999999,-999999);
73  QVector3D Center;
74 
75  for(t = 0; t < triList.size(); t++)
76  {
77  cTri = &triList[t];
78 
79  if(cTri->maxBound.x() > MaxBound.x())
80  MaxBound.setX(cTri->maxBound.x());
81  if(cTri->maxBound.y() > MaxBound.y())
82  MaxBound.setY(cTri->maxBound.y());
83  if(cTri->maxBound.z() > MaxBound.z())
84  MaxBound.setZ(cTri->maxBound.z());
85 
86  if(cTri->minBound.x() < MinBound.x())
87  MinBound.setX(cTri->minBound.x());
88  if(cTri->minBound.y() < MinBound.y())
89  MinBound.setY(cTri->minBound.y());
90  if(cTri->minBound.z() < MinBound.z())
91  MinBound.setZ(cTri->minBound.z());
92  }
93 
94  Center = (MaxBound + MinBound)*0.5;
95 
96  //now we need to move all the triangles...
97  for(t = 0; t < triList.size(); t++)
98  {
99  for(v=0;v<3;v++)
100  {
101  triList[t].vertex[v] -= Center;
102  }
103  triList[t].UpdateBounds(); // since we are moving every triangle, we need to update their bounds too.
104  }
105 }

109  unsigned int t;

251 void B9SupportStructure::ImportAttachmentDataFromStls()
252 {
253  int i;
254  bool s;
255  bool triangleError;
256  STLTri* pLoadedTri = NULL;
257  Triangle3D newtri;
258  QDir appdir(CROSS_OS_GetDirectoryFromLocationTag("APPLICATION_DIR"));
259  QStringList filters;
260  B9SupportAttachmentData importedData;
261 
262  qDebug() << "B9SupportStructure: Importing Stl Attachments...";
263 
264  //Import All Stl in the application directory begining with "SUPPORT"
265  filters << "SUPPORT_*";
266  appdir.setNameFilters(filters);
267  QFileInfoList stlFiles = appdir.entryInfoList();
268  for(i = 0; i < stlFiles.size(); i++)
269  {
270  B9ModelLoader stlLoader(stlFiles[i].filePath(),s);
271  if(s)
272  {
273  while(stlLoader.LoadNextTri(pLoadedTri,triangleError))
274  {
275  if(triangleError)
276  {
277  qDebug() << "B9SupportStructure: ErrorLoading triangle";
278  return;
279  }
280  newtri.normal.setX(pLoadedTri->nx);
281  newtri.normal.setY(pLoadedTri->ny);
282  newtri.normal.setZ(pLoadedTri->nz);
283 
284  newtri.vertex[0].setX(pLoadedTri->x0);
285  newtri.vertex[0].setY(pLoadedTri->y0);
286  newtri.vertex[0].setZ(pLoadedTri->z0);
287  newtri.vertex[1].setX(pLoadedTri->x1);
288  newtri.vertex[1].setY(pLoadedTri->y1);
289  newtri.vertex[1].setZ(pLoadedTri->z1);
290  newtri.vertex[2].setX(pLoadedTri->x2);
291  newtri.vertex[2].setY(pLoadedTri->y2);
292  newtri.vertex[2].setZ(pLoadedTri->z2);
293 
294  delete pLoadedTri;
295  newtri.UpdateBounds();
296 
297  importedData.GetTriangles()->push_back(newtri);
298  }
299 
300  //save the data into the Static List.
301  importedData.SetName(stlFiles[i].baseName().remove("SUPPORT_"));
302  importedData.CenterGeometry();//center the triangles around 0,0,0
303  B9SupportStructure::AttachmentDataList.push_back(importedData);
304  //clear out importedData
305  importedData = B9SupportAttachmentData();
306  }
307  else
308  {
309  qDebug() << "B9SupportStructure: Error Loading: " << stlFiles[i].fileName();
310  }
311 
312  }
313 
314  qDebug() << "B9SupportStructure: Succesfully Loaded "
315  << B9SupportStructure::AttachmentDataList.size()
316  << " Attachment Types";

725 bool B9SupportStructure::IsConstricted(double &depthPastContricted)
726 {
727  double totalHieght;
728  double topHieght;
729  double bottomHiegth;
730 
731  if(IsUnderground(depthPastContricted))
732  {
733  return true;
734  }
735 
736 
737 
738  totalHieght = topPoint.z() - bottomPoint.z();
739  topHieght = fabs(topPoint.z() - topPivot.z());
740  bottomHiegth = fabs(bottomPivot.z() - bottomPoint.z());
741

1223 void B9SupportStructure::ForceUpdate()
1224 {
1225  QVector3D topFinalNormal;//result of combinging angle factor with source normal.
1226  QVector3D bottomFinalNormal;
1227  QVector3D alongMidDownNormal;
1228  QVector3D alongMidUpNormal;
1229  QVector3D topPivotCross;
1230  QVector3D bottomPivotCross;
1231 
1232 
1233  if(isGrounded)
1234  {
1235  bottomPoint.setZ(-instanceParent->GetPos().z());
1236  }
1237 
1238  length = Distance3D(GetTopPoint(),GetBottomPoint());
1239 
1240  //Compute real normal using angle factors
1241  topFinalNormal = topNormal;
1242  topFinalNormal.setX(topNormal.x()*topAngleFactor);
1243  topFinalNormal.setY(topNormal.y()*topAngleFactor);
1244  topFinalNormal.normalize();
1245 
1246  bottomFinalNormal = bottomNormal;
1247  bottomFinalNormal.setX(bottomNormal.x()*bottomAngleFactor);
1248  bottomFinalNormal.setY(bottomNormal.y()*bottomAngleFactor);
1249  bottomFinalNormal.normalize();
1250 
1251 
1252  //Compute pivot positions
1253  topPivot = topPoint - topFinalNormal*topLength;
1254  bottomPivot = bottomPoint - bottomFinalNormal*bottomLength;
1255 
1256 
1257  //Compute midExtensions based on normal angles,
1258  //these extensions are to cover up top and botton shape gaps.
1259  alongMidDownNormal = (bottomPivot - topPivot);
1260  alongMidDownNormal.normalize();
1261  topPivotCross = QVector3D::crossProduct(topFinalNormal,alongMidDownNormal);
1262  topMidExtension = topPivotCross.length()*topRadius;
1263  topMidExtensionPoint = topPivot - alongMidDownNormal*topMidExtension;
1264 
1265  alongMidUpNormal = (topPivot - bottomPivot);
1266  alongMidUpNormal.normalize();
1267  bottomPivotCross = QVector3D::crossProduct(bottomFinalNormal,alongMidUpNormal);
1268  if(isGrounded)
1269  {
1270  bottomMidExtension = midRadius*2*bottomPivotCross.length();
1271  }
1272  else
1273  bottomMidExtension = bottomPivotCross.length()*bottomRadius;
1274 
1275 
1276  bottomMidExtensionPoint = bottomPivot - alongMidUpNormal*bottomMidExtension;
1277 
1278 
1279 
1280  //True rendered mid length
1281  midLength = Distance3D(topMidExtensionPoint,bottomMidExtensionPoint);
1282 
1283 
1284 
1285  //Compute PenetrationPoints (the true end points that are rendered/baked)
1286  topPenetrationPoint = topPoint + topFinalNormal*topPenetration;
1287  bottomPenetrationPoint = bottomPoint + bottomFinalNormal*bottomPenetration;
1288 
1289 
1290  //Compute Euler rotations for top, mid, and bottom shapes
1291  topThetaX = qAcos(topFinalNormal.z())*(180/M_PI);
1292  topThetaZ = qAtan2(topFinalNormal.y(),topFinalNormal.x())*(180/M_PI)-90;
1293 
1294  if(topThetaZ <= 90)
1295  topThetaZ += 180;
1296 
1297  midThetaX = qAcos((topMidExtensionPoint.z() - bottomMidExtensionPoint.z())/midLength)*(180/M_PI);
1298  midThetaZ = qAtan2((topMidExtensionPoint.y() - bottomMidExtensionPoint.y()),
1299  (topMidExtensionPoint.x() - bottomMidExtensionPoint.x()))*(180/M_PI)-90;
1300 
1301  if(midThetaZ <= 90)
1302  midThetaZ += 180;
1303 
1304

1306  bottomThetaZ = qAtan2(bottomFinalNormal.y(),bottomFinalNormal.x())*(180/M_PI)-90;
1307

1309  bottomThetaZ += 180;
