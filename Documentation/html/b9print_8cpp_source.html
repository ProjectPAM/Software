134  int iM = iTime/60000;

316 void B9Print::exposeTBaseLayer(){
317  //Release & reposition cycle completed, time to expose the new layer
318  if(m_iPrintState==PRINT_NO || m_iPrintState == PRINT_ABORT)return;
319 
320  if(m_iPrintState==PRINT_SETUP1){
321  //We've used - getHardZDownMM()to overshoot and get to here,
322  // now reset current position to 0 and move up to + getZFlushMM
323  m_pTerminal->rcResetCurrentPositionPU(0);
324  m_pTerminal->rcBasePrint(m_pTerminal->getZFlushMM());
325  m_iPrintState = PRINT_SETUP2;
326  return;
327  }
328 
329  if(m_iPrintState==PRINT_SETUP2){
330  //We should now be flush
331  // reset current position 0 and continue
332  m_pTerminal->rcResetCurrentPositionPU(0);
333  m_iPrintState = PRINT_RELEASING;
334  }
335 
336  if(m_iPrintState == PRINT_DONE){
337  m_iPrintState=PRINT_NO;
338  m_pTerminal->setEnabled(true);
339  if(m_pTerminal->getPrintPreview()){
340  m_pTerminal->setPrintPreview(false);
341  m_pTerminal->setUsePrimaryMonitor(false);
342  }
343  m_pTerminal->onScreenCountChanged();
344  hide();
345  return;
346  }
347 
348  if(m_bAbort){
349  // We're done, release and raise
350  m_pTerminal->rcSetCPJ(NULL); //blank
351  ui->pushButtonAbort->setText("Abort");
352  m_iPrintState = PRINT_ABORT;
353  on_signalAbortPrint();
354  return;
355  }
356 
357  //Start Tbase Print exposure
358  ui->progressBarPrintProgress->setValue(m_iCurLayerNumber+1);
359  ui->lineEditLayerCount->setText("Creating Layer "+QString::number(m_iCurLayerNumber+1)+" of "+QString::number(m_iLastLayer)+", "+QString::number(100.0*(double)(m_iCurLayerNumber+1)/(double)m_iLastLayer,'f',1)+"% Complete");
360  setSlice(m_iCurLayerNumber);
361  m_vClock.start(); // image is out there, start the clock running!
362  QString sTimeUpdate = updateTimes();
363  if(m_iPaused==PAUSE_WAIT){
364  ui->lineEditLayerCount->setText("Pausing...");
365  setProjMessage("Pausing...");
366  }
367  else{
368  setProjMessage("(Press'p' to pause, 'A' to ABORT) " + sTimeUpdate+" Creating Layer "+QString::number(m_iCurLayerNumber+1)+" of "+QString::number(m_iLastLayer));
369  }
370  m_iPrintState = PRINT_EXPOSING;
371  // set timer
372  int iAdjExposure = m_pTerminal->getLampAdjustedExposureTime(m_iTbase);
373  if(m_iCurLayerNumber<m_iNumAttach) iAdjExposure = m_pTerminal->getLampAdjustedExposureTime(m_iTattach); //First layers may have different exposure timing
374  if(iAdjExposure>0){
375  QTimer::singleShot(iAdjExposure-m_vClock.elapsed(), this, SLOT(startExposeTOverLayers()));
376  return;
377  }
378  else
379  {
380  startExposeTOverLayers(); // If this is getting called, we're taking too long!
381  qDebug() << "EXPOSURE TIMING ERROR: Tbase exposed for too long!, Tbase is set too small or computer too slow?" << iAdjExposure;
382  return;
383  }

386 void B9Print::startExposeTOverLayers(){
387  if(m_iCurLayerNumber==0){exposureOfTOverLayersFinished(); return;} //Skip this on first layer (0)
388 
389  m_vClock.start(); //restart for Tover interval pace
390 
391  m_iMinimumTintMS = m_vSettings.value("m_iMinimumTintMS",50).toInt(); // We default to 50ms but adjust it upwards when it gets hit.
392  m_iMinimumTintMSWorstCase=m_iMinimumTintMS;
393

403 void B9Print::exposureOfCurTintLayerFinished(){
