75  double resy;

79  bool flipped = false;
80  int numSupports;
81  int hasFoundation;
82 
83  QFile file(filepath);
84  QTextStream in;
85  QString buff;//for io operations
86  QString modelpath;//current model this function is trying to load.
87 
88 
89  //input file operations here
90  if(!file.open(QIODevice::ReadOnly))
91  {
92  return false;
93  }
94  //check file extension
95  if(QFileInfo(filepath).completeSuffix().toLower() != "b9l")
96  return false;
97 
98 
99  SetDirtied(false);//presumably good from here so undirty the project
100 
101  in.setDevice(&file);//begin text streaming.
102  in >> buff;
103  if(buff == "ver")//we are looking at a post 1.4 b9 layout file
104  {
105  in >> buff;
106  FileVersion = buff.toInt();//should be 14 or greater.
107  in.skipWhiteSpace();
108  mfilename = in.readLine();//get project name
109  }
110  else//old file - we need to read first line for project name
111  {
112  FileVersion = 13;
113 
114  in.seek(0);//go back and re-read.
115  mfilename = in.readLine();//get project name
116  }
117 
118  qDebug() << "Layout Version: " << FileVersion;
119 
120  buff = in.readLine();//eat startmodellist
121  while(!jumpout)
122  {
123  in.skipWhiteSpace();
124  buff = in.readLine();
125 
126  if(buff != "endinstancelist")
127  {
128  modelpath = buff;
129 
130  if(!QFileInfo(modelpath).exists())
131  {
132  modelpath = QFileInfo(filepath).absolutePath() + "/" + QFileInfo(modelpath).fileName();
133  }
134 
135  B9ModelInstance* pinst = pMain->AddModel(modelpath,withoutVisuals);//inst will be null if it cant find the file.
136 
137  //model relocation chanches
138  if(pinst == NULL)
139  {
140  PromptFindLostModel(pinst,modelpath);
141  }
142 
143 
144  in >> xp >> yp >> zp;
145  in >> xr >> yr >> zr;
146  in >> xs >> ys >> zs;
147 
148  //version 14 added instance flipping
149  if(FileVersion >= 14)
150  {
151  in >> buff;
152  flipped = (bool)buff.toInt();
153  }
154  else
155  flipped = false;
156 
157  if(pinst != NULL)
158  {
159  pinst->SetPos(QVector3D(xp,yp,zp));
160  pinst->SetRot(QVector3D(xr,yr,zr));
161  pinst->SetScale(QVector3D(xs,ys,zs));
162  pinst->SetFlipped(flipped);
163  pinst->UpdateBounds();//in order that zhieght is good and all.
164  }
165  //version 15 added instance supports
166  // add the supports to the loaded model
167  if(FileVersion >= 15)
168  {
169  in.skipWhiteSpace();
170  in >> buff;//eat "beginsupports"
171  in >> numSupports;
172  in >> hasFoundation;
173 
174  if(hasFoundation)
175  StreamInSupportInfo(pinst,in,true);
176 
177  for(i = 0; i < numSupports; i++)
178  {
179  StreamInSupportInfo(pinst,in,false);
180  }
181  in >> buff; in.skipWhiteSpace();//eat "endsupports"
182  }
183  }
184  else
185  {
186  jumpout = true;
187  }
188  }
189 
190  in >> buff >> resx >> resy;
191  in >> buff >> xypixel;
192 
193 
194  SetResolution(QVector2D(resx,resy));
195  CalculateBuildArea();
196  mfilename = filepath;
197 
198 
199  //some post load warnings for legacy layouts
200  if(FileVersion < 15)
201  {
202  QMessageBox msgBox;
203  msgBox.setIcon(QMessageBox::Warning);
204  msgBox.setText("This Layout was created with an older version of b9creator");
205  msgBox.setInformativeText("Some models may be rotated differently");
206  msgBox.exec();
207  }
208 
209  Disable_User_Waiting_Cursor();
210 
211  emit ProjectLoaded();
212  return true;
213  }

216  bool B9LayoutProjectData::Save(QString filepath) //returns success

244  out << "beginsupports " << inst->GetSupports().size() << " " << instHasFoundation << "\n";
245 
246  //DEFINiTION: if there is a foundation support, its the first one.
247  if(instHasFoundation)
248  StreamOutSupportInfo(inst->GetBasePlate(),out);
249 
250  for(j = 0; j < inst->GetSupports().size(); j++)
251  {
252  pSup = inst->GetSupports()[j];
253  StreamOutSupportInfo(pSup,out);
254  }
255  out << "endsupports" << "\n";
256  }
257 
258 
259 
260  out << "endinstancelist" << '\n';//end model list
261 
262  out << "resolution " << GetResolution().x() << " " << GetResolution().y() << '\n';
263  out << "XYpixelsize " << GetPixelSize() << '\n';
264 
265 
266  SetDirtied(false);
267  mfilename = filepath;
268  return true;
