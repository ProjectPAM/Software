75 bool SegmentIntersection(QVector2D &result, QVector2D seg11, QVector2D seg12, QVector2D seg21, QVector2D seg22)
76 {
77  // Store the values for fast access and easy
78  // equations-to-code conversion
79  double x1 = seg11.x(), x2 = seg12.x(), x3 = seg21.x(), x4 = seg22.x();
80  double y1 = seg11.y(), y2 = seg12.y(), y3 = seg21.y(), y4 = seg22.y();
81 
82  double d = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
83  // If d is zero, there is no intersection
84  if (d == 0) return NULL;
85 
86  // Get the x and y
87  double pre = (x1*y2 - y1*x2), post = (x3*y4 - y3*x4);
88  double x = ( pre * (x3 - x4) - (x1 - x2) * post ) / d;
89  double y = ( pre * (y3 - y4) - (y1 - y2) * post ) / d;
90 
91  // Check if the x and y coordinates are within both lines
92  if ( x < std::min(x1, x2) || x > std::max(x1, x2) ||
93  x < std::min(x3, x4) || x > std::max(x3, x4) ) return false;
94  if ( y < std::min(y1, y2) || y > std::max(y1, y2) ||
95  y < std::min(y3, y4) || y > std::max(y3, y4) ) return false;
96 
97  // Return the point of intersection
98  result.setX(x);
99  result.setY(y);
100  return true;
101 }
