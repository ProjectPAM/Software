951 int B9Terminal::getEstCompleteTimeMS(int iCurLayer, int iTotLayers, double dLayerThicknessMM, int iExposeMS)
952 {
953  //return estimated completion time
954  int iTransitionPointLayer = (int)(pSettings->m_dBTClearInMM/dLayerThicknessMM);
955

957  int iUpperCount = iTotLayers - iLowerCount;
958

960  if(iUpperCount<0)iUpperCount=0;
961 
962  if(iCurLayer<iTransitionPointLayer)iLowerCount = iLowerCount-iCurLayer; else iLowerCount = 0;
963  if(iCurLayer>=iTransitionPointLayer) iUpperCount = iTotLayers - iCurLayer;
964 
965  int iTotalTimeMS = iExposeMS*iLowerCount + iExposeMS*iUpperCount;
966 
967  iTotalTimeMS = getLampAdjustedExposureTime(iTotalTimeMS);
968 
969  // Add Breathe and Settle
970  iTotalTimeMS += iLowerCount*(pSettings->m_dBreatheClosed1 + pSettings->m_dSettleOpen1)*1000;
971  iTotalTimeMS += iUpperCount*(pSettings->m_dBreatheClosed2 + pSettings->m_dSettleOpen2)*1000;
972

974  int iGap1 = iLowerCount*(int)(pSettings->m_dOverLift1*100000.0/(double)pPrinterComm->getPU());
975  int iGap2 = iUpperCount*(int)(pSettings->m_dOverLift2*100000.0/(double)pPrinterComm->getPU());
976

981  int iZLowerDistance2 = iGap2;
982 
983  iTotalTimeMS += getZMoveTime(iZRaiseDistance1,pSettings->m_iRSpd1);
984  iTotalTimeMS += getZMoveTime(iZRaiseDistance2,pSettings->m_iRSpd2);
985  iTotalTimeMS += getZMoveTime(iZLowerDistance1,pSettings->m_iLSpd1);
986  iTotalTimeMS += getZMoveTime(iZLowerDistance2,pSettings->m_iLSpd2);
